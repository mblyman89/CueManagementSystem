# GPIO configuration for Raspberry Pi Zero W2
# Add this to /boot/config.txt

# Set GPIO 2, 3, 4, 5, and 6 to HIGH at boot
# gpio=2,3,4,5,6=op,dh

# Set GPIO 9 to LOW at boot with pull-down
# gpio=9=op,dl,pd

# Alternative method using gpio-hog overlay
dtoverlay=gpio-hog,gpio2=op,dh,gpio3=op,dh,gpio4=op,dh,gpio5=op,dh,gpio6=op,dh,gpio9=op,dl,pd

# Note: op = output, dh = drive high, dl = drive low, pd = pull-down
